function getPatternResult($pattern,$str) {
	$re='';
	if(preg_match($pattern,$str,$re)){
		//使用的时候这里注释掉就行了
		echo $re[1];
		return $re[1];
	}
	else{
		//使用的时候这里注释掉就行了
		echo "error";
		return "error";
	}
}
